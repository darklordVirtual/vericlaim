# Faster code

def solve(amount):

coins = [200, 300]

prices = [380, 550]

dp = [float(’inf’)] * (amount + 1)

dp[0] = 0

for i in range(len(coins)):

for j in range(coins[i], amount+1):

dp[j] = min(dp[j], dp[j - coins[i]] + prices[i])

return dp[amount]
